# raspi_tensorflow_install
## How to install
	$ sudo apt-get update
	$ sudo apt-get upgrade
	$ git clone https://github.com/ft-jasong/raspi_tensorflow_install.git && cd raspi_tensorflow_install
	$ ./pre_install.sh
	$ ./pyenv_install.sh
	$ ./env_setting.sh
	$ pyenv --version
	$ pyenv install 3.7.9
	$ pyenv global 3.7.9
	$ pip3 install tensorflow
